CREATE TABLE "TODOLIST"
("TODO_NUMBER" NUMBER PRIMARY KEY,
"TODO_CONTENT" VARCHAR2(1000),
"TODO_ST" CHAR(1) DEFAULT 'N' NOT NULL,
"MEMBER_NO" NUMBER NOT NULL);

COMMENT ON COLUMN "TODOLIST"."TODO_NUMBER" IS '투두리스트 번호(시퀀스)';
COMMENT ON COLUMN "TODOLIST"."TODO_CONTENT" IS '투두리스트 내용';
COMMENT ON COLUMN "TODOLIST"."TODO_ST" IS '투두리스트 삭제여부';
COMMENT ON COLUMN "TODOLIST"."MEMBER_NO" IS '투두리스트 작성자번호';

CREATE SEQUENCE SEQ_TODO_NO;
